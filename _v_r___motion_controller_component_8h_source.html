54template<class filterType>
55class FBasicLowPassFilter
56{
57public:
58
60 FBasicLowPassFilter(filterType EmptyValueSet)
61 {
62 EmptyValue = EmptyValueSet;
63 Previous = EmptyValue;
64 PreviousRaw = EmptyValue;
65 bFirstTime = true;
66 }
67
69 filterType Filter(const filterType& InValue, const filterType& InAlpha)
70 {
71
72 filterType Result = InValue;
73 if (!bFirstTime)
74 {
75 // This is unsafe in non float / float array data types, but I am not going to be using any like that
76 for (int i = 0; i < sizeof(filterType) / sizeof(double); i++)
77 {
78 ((double*)&Result)[i] = ((double*)&InAlpha)[i] * ((double*)&InValue)[i] + (1.0f - ((double*)&InAlpha)[i]) * ((double*)&Previous)[i];
79 }
80 }
81
82 bFirstTime = false;
83 Previous = Result;
84 PreviousRaw = InValue;
85 return Result;
86 }
87
88 filterType EmptyValue;
89
91 filterType Previous;
92
94 filterType PreviousRaw;
95
97 bool bFirstTime;
98
99 //private:
100
101 const filterType CalculateCutoff(const filterType& InValue, double& MinCutoff, double& CutoffSlope)
102 {
103 filterType Result;
104 // This is unsafe in non float / float array data types, but I am not going to be using any like that
105 for (int i = 0; i < sizeof(filterType) / sizeof(double); i++)
106 {
107 ((double*)&Result)[i] = MinCutoff + CutoffSlope * FMath::Abs(((double*)&InValue)[i]);
108 }
109 return Result;
110 }
111
112 const filterType CalculateAlpha(const filterType& InCutoff, const double InDeltaTime)
113 {
114 filterType Result;
115 // This is unsafe in non float / float array data types, but I am not going to be using any like that
116 for (int i = 0; i < sizeof(filterType) / sizeof(double); i++)
117 {
118 ((double*)&Result)[i] = CalculateAlphaTau(((double*)&InCutoff)[i], InDeltaTime);
119 }
120 return Result;
121 }
122
123 inline const double CalculateAlphaTau(const double InCutoff, const double InDeltaTime)
124 {
125 const double tau = 1.0 / (2.0 * PI * InCutoff);
126 return 1.0 / (1.0 + tau / InDeltaTime);
127 }
128};
