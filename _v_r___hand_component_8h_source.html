875 int32 GetBoneIndexHelper(FName InBoneName, const USkeletalMeshComponent& SkelComp, int32* BodyIndex = nullptr) const
876 {
877 FName BoneName = InBoneName;
878 const UPhysicsAsset* PhysAsset = SkelComp.GetPhysicsAsset();
879
880 if (BoneName == NAME_None)
881 {
882 //Didn't specify bone name so just use root body
883 if (PhysAsset)
884 {
885 const int32 RootBodyIndex = SkelComp.FindRootBodyIndex();
886 if (PhysAsset->SkeletalBodySetups.IsValidIndex(RootBodyIndex))
887 {
888 BoneName = PhysAsset->SkeletalBodySetups[RootBodyIndex]->BoneName;
889 }
890 }
891 }
892
893 if (BodyIndex)
894 {
895 *BodyIndex = PhysAsset ? PhysAsset->FindBodyIndex(BoneName) : INDEX_NONE;
896 }
897
898 return SkelComp.GetBoneIndex(BoneName);
899 }
